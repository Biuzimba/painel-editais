    <script>
        // Banco de dados local simples
        let pacientes = JSON.parse(localStorage.getItem('pacientes')) || [];
        let agenda = JSON.parse(localStorage.getItem('agenda')) || [];

        function showTab(tabId) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.classList.remove('active'));
            document.getElementById(tabId).classList.add('active');
            renderizar();
        }

        function toggleModal(id) {
            const modal = document.getElementById(id);
            modal.classList.toggle('hidden');
            
            // Popular select de pacientes no agendamento
            if(id === 'modal-agenda') {
                const select = document.getElementById('a-paciente');
                select.innerHTML = pacientes.map(p => `<option value="${p.nome}">${p.nome}</option>`).join('');
            }
        }

        function salvarPaciente() {
            const nome = document.getElementById('p-nome').value;
            const queixa = document.getElementById('p-queixa').value;
            if(!nome) return alert("Nome é obrigatório");

            pacientes.push({ id: Date.now(), nome, queixa });
            localStorage.setItem('pacientes', JSON.stringify(pacientes));
            toggleModal('modal-paciente');
            renderizar();
        }

        function agendarSessao() {
            const paciente = document.getElementById('a-paciente').value;
            const data = document.getElementById('a-data').value;
            const hora = document.getElementById('a-hora').value;

            if(!data || !hora) return alert("Preencha data e hora");

            agenda.push({ id: Date.now(), paciente, data, hora, status: 'Confirmado' });
            localStorage.setItem('agenda', JSON.stringify(agenda));
            toggleModal('modal-agenda');
            renderizar();
        }

        function renderizar() {
            // Stats
            document.getElementById('count-pacientes').innerText = pacientes.length;
            const hoje = new Date().toISOString().split('T')[0];
            document.getElementById('count-hoje').innerText = agenda.filter(a => a.data === hoje).length;

            // Lista Pacientes
            const grid = document.getElementById('grid-pacientes');
            grid.innerHTML = pacientes.map(p => `
                <div class="bg-white p-5 rounded-xl border border-slate-100 shadow-sm relative group">
                    <h4 class="font-bold text-slate-800">${p.nome}</h4>
                    <p class="text-sm text-slate-500 mt-2 italic">"${p.queixa}"</p>
                    <div class="mt-4 flex gap-2">
                         <span class="text-[10px] bg-blue-50 text-blue-600 px-2 py-1 rounded-full uppercase font-semibold">Paciente Ativo</span>
                    </div>
                </div>
            `).join('');

            // Tabela Agenda
            const tabela = document.getElementById('lista-agenda');
            tabela.innerHTML = agenda.sort((a,b) => a.data.localeCompare(b.data)).map(a => `
                <tr class="border-b hover:bg-slate-50 transition">
                    <td class="p-4 font-medium">${a.paciente}</td>
                    <td class="p-4">${a.data.split('-').reverse().join('/')}</td>
                    <td class="p-4">${a.hora}</td>
                    <td class="p-4"><span class="bg-emerald-100 text-emerald-700 text-xs px-2 py-1 rounded-full font-bold">${a.status}</span></td>
                </tr>
            `).join('');
        }

        // Inicializar
        renderizar();
    </script>
